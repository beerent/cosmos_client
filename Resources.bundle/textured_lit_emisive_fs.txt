#version 100 
precision highp float;
varying vec2 fTexCord;
varying vec3 fNormal;
uniform sampler2D mytexture;
uniform sampler2D myEmisivetexture;
void main(void) {       
gl_FragColor = texture2D(mytexture, fTexCord  * vec2(1.0, -1.0)) * vec4(fNormal, 1.0) + texture2D(myEmisivetexture, fTexCord * vec2(1.0, -1.0));
}
